                  class  Subject{
        constructor(){
            this.callbacks= []
        }
        subscribe(fn){
            if(this.callbacks.indexOf(fn) === -1){
                 this.callbacks.push(fn);
            }      
        }
        unsubscribe(fn){
            this.callbacks.filter(callback => callback !== fn)
        }

        publish(msg){
            this.callbacks.forEach(fn=>fn(msg))
        }

    }

              # 2 观察者模式

                 class Dep {
       constructor(){
         this.state = null;
         this.observers = []
        }

        subscribe(fn){
            if(this.observers.indexOf(fn) === -1){
                 this.observers.push(fn);
            }      
        }

        unsubscribe(fn){
            this.observers.filter(callback => callback !== fn)
        }

        publish(msg){
            this.observers.forEach(observer=>observer.update(msg))
        }

        setState(state){
            this.state = state;
            this.publish(state);
        }
   };

   class Observer{
       constructor(name){
         this.name = name;
       }
       update(msg){
          console.log(`${this.name} 观察到 ${msg}`)     
       }
   }

   const dep = new Dep();
   const observer1 = new Observer('观察者1号');
   const observer2 = new Observer('观察者2号');

   dep.subscribe(observer1);
   dep.subscribe(observer2);

   dep.setState('彭时宇有一条新的动态')

              # 4 单例模式

                     function singleton(fn){
        var result;
        return function(){
            return result || (result = fn.call(this, arguments))
        }
       }

              # 5 代理模式

                    class Girl {
          constructor(name){
            this.name = name;
          }
      }

      class Boy {
        constructor(girl){
            this.girl = girl;
        }
        sendEmail(emal){
            console.log(`你好${this.girl}, ${emal}`)
        }
      }

      class ProxyObj {
        constructor(girl,boy){
            this.girl = girl;
            this.boy = boy
        }
        sendMessage(message){
            this.boy.sendEmail(message)
        }
      }

     const ChangE = new Girl('嫦娥');
     const ZhuBaJie = new Boy('猪八戒');

     const xiaoming = new ProxyObj(ChangE, ZhuBaJie);
     xiaoming.sendMessage('今晚一起看电影好吗');

              # 6 依赖注入

              // 建立一个存储服务的容器
       const deps = {
          eat: function(){
            console.log('吃饭饭');
          };
          drink: function(){
            console.log('喝茶茶');
          }
       }
 // 创建一个依赖注入的函数
  class DI{
       constructor(dependency){
          this.dependency = dependency;
       }

       inject(fn){
         const result = fn.toString().match(/^[^(]+\(([^)]+)\)/);
         const params = result ? result[1].split(',') : [];
         const paramsdeps = params.map(dep => this.dependency[dep]);

         return function(){
            return fn.apply(null,paramsdeps);
         }
       }
  }

  const di = new DI(deps);
  const play = di.inject(function(eat,drink){
    eat();
    drink();
  })
