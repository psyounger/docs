        # 1节流

            //方法一
    function throttle1(fn, delay){
        let flag = false;
        return function(){
            if(flag) return;
            flag = true;
            setTimeout(()=>{
                fn();
                flag = false;
            },delay);
        }
    }

   //方法二
    function throttle2(fn, delay){
        let firstTime = Date.now();
        return function(){
            let nowTime = Date.now();
            if(nowTime - firstTime < delay ) return;
            fn();
            firstTime = nowTime;
        }
    }

        # 2防抖

            function antiShake(fn, delay){
        let timer = null;
        return function(){
            clearTimeout(timer);
            timer = setTimeout(()=>{
                fn();
            },delay)
        }
    }

            //方法一 冒泡排序
    function bubblingSort(arr){
        for( const i = 0;  i < arr.length -1; i++;){
              for(const j = 0; j < arr.length -1 - i; j++; ){
                   if(arr[j] > arr[j+1] ){
                    //数组结构赋值
                      [ arr[j], arr[j+1] ] = [ arr[j+1], arr[j] ];
                   }
              }
        }
        return arr;
    };
    //方法二 选择排序
    function selectionSort(arr){
        for( const i = 0; i < arr.length -1;  i++;){
              for(const j = i+1;  j < arr.length -1; j++;){
                   if(arr[i] > arr[j] ){
                    //数组结构赋值
                      [ arr[i], arr[j] ] = [ arr[j], arr[i] ];
                   }
              }
        }
        return arr;
    };

    function quickSort1(arr){
            if(arr.length <= 1) return arr;
         const middleIndex = Math.floor(arr.length/2);
         const left = [];
         const right = [];
         middleValue = arr.splice(middleIndex,1)[0]

         for(const i = 0; i++; i < arr.length ){
            if(middleValue >= arr[i]){
                left.push(arr[i])
            }else{
                right.push(arr[i])
            }
         }
          return [ ...quickSort(left), middleValue, ...quickSort(right)];
       };

       function quickSort2(arr){
       
        if(arr.length<=1) return arr;

        let flag = arr.shift();//移除数组中的第一项并返回
        let left = [];
        let right = [];
        for (var i = 0; i<arr.length; i++){
            if(flag > arr[i]){
             left.push(arr[i]);
            }else{
                right.push(arr[i])
            }
        }
        return [...quickSort(left),flag,...quickSort(right)];
